{{- $resource := 0 -}} {{- $url := urls.Parse .Path -}} {{- if not $url.Host | and $url.Path | and (strings.HasSuffix $url.Path "/" | not) -}} {{- if .Resources -}} {{- with .Resources.GetMatch $url.Path -}} {{- $resource = . -}} {{- end -}} {{- end -}} {{- if not $resource -}} {{- with resources.Get $url.Path -}} {{- $resource = . -}} {{- end -}} {{- end -}} {{- end -}} {{- return $resource -}}
